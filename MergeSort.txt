#include <iostream>
#include <sys/time.h>
#include <time.h>
#include <stdio.h>
#include <stdlib.h>

using namespace std;



void merge(int arr[], int lft, int mid, int rht)
{
	int i, j, k;
	int n1ft = mid - lft + 1;
	int nrht = rht - mid;

	// create temp arrays
	int L[n1ft], R[nrht];


	for (i = 0; i < n1ft; i++)
		L[i] = arr[lft + i];
	for (j = 0; j < nrht; j++)
		R[j] = arr[mid + 1+ j];

	/* Merge the temp arrays back into arr[l..r]*/
	i = 0; // Initial index of first subarray
	j = 0; // Initial index of second subarray
	k = lft; // Initial index of merged subarray
	while (i < n1ft && j < nrht)
	{
		if (L[i] <= R[j])
		{
			arr[k] = L[i];
			i++;
		}
		else
		{
			arr[k] = R[j];
			j++;
		}
		k++;
	}



	while (i < n1ft)
	{
		arr[k] = L[i];
		i++;
		k++;
	}


	while (j < nrht)
	{
		arr[k] = R[j];
		j++;
		k++;
	}
}


void mergeSort(int arr[], int l, int r)
{
	if (l < r)
	{
		int m = l+(r-l)/2;
		mergeSort(arr, l, m);
		mergeSort(arr, m+1, r);

		merge(arr, l, m, r);
	}
}


void showArray(int A[], int size)
{
	int i;
	for (i=0; i < size; i++)
		cout<<A[i]<<" ";
}


int main()
{
    int n;
    cout<<"Enter number of elements: "<<endl;

    int arr[n];
    cout<<"Enter the elements: "<<endl;
    for(int i=0; i<n; i++)
        cin>>arr[i];


	int arr_size = sizeof(arr)/sizeof(arr[0]);

	printf("Given array is \n");
	showArray(arr, arr_size);

	mergeSort(arr, 0, arr_size - 1);

	cout<<"\nSorted array is \n";
	showArray(arr, arr_size);
	return 0;
}
